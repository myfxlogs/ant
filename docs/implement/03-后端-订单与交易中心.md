## 1. MtHubService 结构

```go
type MtHubService struct {
    hub            *Hub                  // MT 会话注册中心（accountID → Session + Executor）
    broker         *OrderEventBroker     // 订单事件总线（per-user pub/sub）
    accountBroker  *AccountProfitBroker  // 账户利润推送（per-account pub/sub）
    snapshotBroker *PositionSnapshotBroker // 持仓快照推送（per-account pub/sub）
    idem           *IdempotencyGuard     // Redis 幂等守卫（24h TTL）
    reconcileGate  *ReconcileGate        // 对账门控（对账中拒绝新订单）
    eventStore     *TradeEventStore      // NATS JetStream 事件持久化
    userLimiter    *usermgr.UserLimiter  // 用户频率限制
    costEstimator  costsvc.CostEstimator // 预交易成本估算
    killSwitch     KillSwitchGate        // 全局熔断
    riskPipeline   RiskPipeline          // 风控管线
    accountStateProvider AccountStateProvider // 账户状态提供者
    omsWriter      *OmsWriter            // OMS 15-state 状态机
}
```

## 2. PlaceOrder 精确流程

```go
func (s *MtHubService) PlaceOrder(ctx, req *OrderRequest) (*OrderRecord, error) {
    // 1. 全局熔断检查
    if s.killSwitch != nil && s.killSwitch.IsEngaged() {
        return nil, ErrKillSwitchEngaged
    }

    // 2. 幂等性检查（Redis, 24h TTL）
    if s.idem != nil && req.ClientID != "" {
        isDup, existingTicket, err := s.idem.CheckAndSet(ctx, accountID, clientID, 0)
        if isDup {
            return &OrderRecord{Ticket: existingTicket, State: OrderStatePending}, ErrDuplicateOrder
        }
    }

    // 3. 对账门控
    if s.reconcileGate != nil && !s.reconcileGate.CanAccept(accountID) {
        return nil, ErrReconciling
    }

    // 4. 用户频率限制（10 订单/秒）
    if s.userLimiter != nil {
        uid := usermgr.GetUserID(ctx)
        if uid != "" && !s.userLimiter.AllowOrder(uid) {
            return nil, ErrRateLimited
        }
    }

    // 5. OMS 插入（state=NEW）
    orderID := IdempotencyKey(accountID, clientID)
    if s.omsWriter != nil {
        s.omsWriter.InsertOrder(ctx, orderID, accountID, canonical,
            int16(orderType), volume, price, sl, tp)
    }

    // 6. 风控管线
    if s.riskPipeline != nil {
        balance, equity, freeMargin, margin := fetchAccountState(...)
        result := s.riskPipeline.Process(ctx, &SignalRequest{
            UserID, AccountID, Symbol, Side, Price,
            Balance, Equity, FreeMargin, Margin, Positions,
        })
        if !result.Allowed {
            s.omsWriter.Transition(orderID, NEW, REJECTED)
            return nil, fmt.Errorf("risk rejected at %s: %s", result.Stage, result.Reason)
        }
        s.omsWriter.Transition(orderID, NEW, VALIDATED)
        s.omsWriter.Transition(orderID, VALIDATED, RISK_APPROVED)

        // 风控 sizer 可能修改 volume
        if result.Lots > 0 { req.Volume = result.Lots }
    } else {
        s.omsWriter.Transition(orderID, NEW, VALIDATED)
        s.omsWriter.Transition(orderID, VALIDATED, RISK_APPROVED)
    }

    // 7. 预交易成本估算
    if s.costEstimator != nil {
        est := s.costEstimator.Estimate(ctx, params)
        costJSON = json.Marshal(est)
    }

    // 8. MT session 执行
    exec := s.hub.Get(accountID)
    if exec == nil {
        s.omsWriter.Transition(orderID, RISK_APPROVED, FAILED)
        return nil, ErrSessionNotFound
    }
    ticket, err := exec.PlaceOrder(ctx, req)
    if err != nil {
        s.omsWriter.Transition(orderID, RISK_APPROVED, FAILED)
        return nil, err
    }
    s.omsWriter.Transition(orderID, RISK_APPROVED, SUBMITTED)

    // 9. 更新幂等键
    if s.idem != nil && req.ClientID != "" {
        s.idem.SetTicket(ctx, accountID, clientID, ticket)
    }

    // 10. 发布 ORDER_CREATED 事件到 NATS JetStream
    s.eventStore.Publish(ctx, &TradeEvent{
        EventType: TradeEventOrderCreated,
        AccountID, Ticket, ClientID, Canonical,
        Side, OrderType, Volume, Price, SL, TP,
        ToState: "SUBMITTED", FromState: "RISK_APPROVED",
        Version: 1, CostBreakdownJSON: costJSON,
    })

    return &OrderRecord{Ticket: ticket, State: OrderStatePending}, nil
}
```

## 5. 事件总线

| Broker | 类型 | 订阅粒度 | 用途 |
|--------|------|----------|------|
| `OrderEventBroker` | Pub/Sub | per-user | 订单状态变更 → SSE 推送 |
| `AccountProfitBroker` | Pub/Sub | per-account | 余额/权益实时更新 → SSE |
| `PositionSnapshotBroker` | Pub/Sub | per-account | 全量持仓快照 → SSE |

订阅模式（以 AccountProfitBroker 为例）：

```go
func (s *MtHubService) SubscribeAccountProfit(ctx, accountID) (<-chan *AccountProfitEvent, func()) {
    return s.accountBroker.Subscribe(accountID)
}

func (s *MtHubService) PublishAccountProfit(ev *AccountProfitEvent) {
    s.accountBroker.Publish(ev)
}
```

## 6. OMS 15-State 状态机

```go
type OmsWriter struct {
    pool       *pgxpool.Pool
    eventStore *TradeEventStore
}
```

状态流转：

```
NEW → VALIDATED → RISK_APPROVED → SUBMITTED
                                     ↓
                                PARTIALLY_FILLED → FILLED
                                     ↓
                                CANCELLED / REJECTED / FAILED / EXPIRED
```

接口：

```go
func (w *OmsWriter) InsertOrder(ctx, orderID, accountID, symbol string,
    orderType int16, volume, price, sl, tp float64) error

func (w *OmsWriter) Transition(ctx, orderID string, from, to OMSState) error
```
